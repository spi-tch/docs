                ```typescript
                import Spitch from 'spitch';
                import fs from 'fs';

                const client = new Spitch({ apiKey: process.env.SPITCH_API_KEY });

                async function speechToText(): Promise<void> {
                    const selectedFile = new File([""], "audio.wav");
                    const response = await client.speech.transcribe({
                        content: selectedFile,
                        language: "en" as const,
                        model: "mansa_v1" as const,
                        special_words: "Chinedu, Lekki",
                        timestamp: "sentence" as const,
                    });
                    console.log("Transcript:", response.text);
                }

                speechToText();
                ```
